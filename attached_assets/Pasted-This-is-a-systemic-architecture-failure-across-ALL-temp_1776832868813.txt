export const getTemplatesByType = (
  type: 'invoice' | 'quotation' | 'receipt'
) => Object.values(TEMPLATE_REGISTRY)
     .filter(t => t.type === type)

// Helper to get a single template:
export const getTemplate = (id: string) => 
  TEMPLATE_REGISTRY[id]

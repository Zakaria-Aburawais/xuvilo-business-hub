  const hasBrandInDesc = KNOWN_BRANDS.some(b =>

  );

  // TYPE 1: Manufacturer + part number — highest confidence
  if ((item.manufacturer || hasBrandInDesc) && item.part_no) {
    return {
      type: 'branded_with_part',
      confidence: 'high',
      strategy: 'manufacturer_part_lookup'
    };
  }

  // TYPE 2: Brand/model in description, no part number
  if (item.manufacturer || hasBrandInDesc) {
    return {
      type: 'branded_no_part',
      confidence: 'medium',
      strategy: 'brand_model_lookup'
    };
  }

  // TYPE 3: Part/stock number only, no manufacturer
  if (item.part_no) {
    return {
      type: 'part_number_only',
      confidence: 'medium',
      strategy: 'part_number_lookup'
    };
  }

  // TYPE 4: Description only
  return {
    type: 'description_only',
    confidence: 'low',
    strategy: 'generic_search'
  };
}

5.6 Image Fetcher — imageFetcher.js
Waterfall image search — try each level in order, stop at first success:
javascriptasync function fetchItemImage(item, researchResult) {

  function isDirectImageUrl(url) {
    if (!url) return false;
    return /\.(jpg|jpeg|png|webp|gif)(\?.*)?$/i.test(url);
  }

  // Level 1: Exact part on manufacturer site
  if (item.manufacturer && item.part_no &&
      researchResult.image_url &&
      isDirectImageUrl(researchResult.image_url)) {
    return {
      url: researchResult.image_url,
      confidence: 'exact',
      badge: '🟢 Exact match',
      source: researchResult.image_source
    };
  }

  // Level 2: Brand match from official site
  if (item.manufacturer &&
      researchResult.image_url &&
      isDirectImageUrl(researchResult.image_url)) {
    return {
      url: researchResult.image_url,
      confidence: 'brand_match',
      badge: '🟡 Brand match',
      source: researchResult.image_source
    };
  }

  // Level 3: Distributor image
  if (researchResult.image_url &&
      isDirectImageUrl(researchResult.image_url)) {
    return {
      url: researchResult.image_url,
      confidence: 'distributor',
      badge: '🟡 Distributor image',
      source: researchResult.image_source
    };
  }

  // Level 4: Representative image
  if (researchResult.image_url) {
    return {
      url: researchResult.image_url,
      confidence: 'representative',
      badge: '🟠 Representative',
      source: researchResult.image_source || 'web'
    };
  }

  // Level 5: No image — category icon fallback
  return {
    url: null,
    confidence: 'none',
    badge: '🔴 No image found',
    source: null,
    icon: getCategoryIcon(item.description)
  };
}

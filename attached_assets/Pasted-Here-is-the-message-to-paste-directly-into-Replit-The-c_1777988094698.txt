function extractKeywords(item) {
  const keywords = [];
  
  // Add manufacturer if present
  if (item.manufacturer) keywords.push(item.manufacturer.toLowerCase());
  
  // Add part number fragments
  if (item.part_no) keywords.push(item.part_no.toLowerCase());
  
  // Extract meaningful words from description
  // Remove common words, keep product-specific terms
  const stopWords = ['the','and','for','with','per','from','type',
                     'unit','each','set','lot','new','original'];
  const descWords = item.description.toLowerCase()
    .split(/[\s,\/\(\)]+/)
    .filter(w => w.length > 2 && !stopWords.includes(w));
  keywords.push(...descWords);
  
  return [...new Set(keywords)]; // deduplicate
}
Step 2 — Query the knowledge base for each item's keywords:

sql
SELECT *, 
  (
    CASE WHEN LOWER(specialties) LIKE '%' || :kw1 || '%' THEN 2 ELSE 0 END +
    CASE WHEN LOWER(specialties) LIKE '%' || :kw2 || '%' THEN 2 ELSE 0 END +
    CASE WHEN LOWER(name)        LIKE '%' || :kw1 || '%' THEN 1 ELSE 0 END +
    CASE WHEN is_local = 1 THEN 1 ELSE 0 END
  ) AS relevance_score
FROM rfq_suppliers
WHERE active = 1
AND relevance_score > 0
ORDER BY relevance_score DESC, is_local DESC
LIMIT 5

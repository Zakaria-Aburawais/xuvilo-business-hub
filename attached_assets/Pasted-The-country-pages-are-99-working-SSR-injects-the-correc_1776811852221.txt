Then in CountryPage.tsx, get the slug like this:

import { useRoute } from 'wouter'

export default function CountryPage() {
  const [match, params] = useRoute(
    '/invoice-generator-:slug'
  )
  const slug = params?.slug
  
  if (!slug) return <NotFound />
  
  const country = countries.find(c => c.slug === slug)
  if (!country) return <NotFound />
  
  return <CountryPageContent country={country} />
}

OPTION B — Read slug from window.location directly
(use this if useRoute still fails):

export default function CountryPage() {
  const slug = window.location.pathname
    .split('/invoice-generator-')[1]
    ?.split('/')[0]
    ?.split('?')[0]
  
  const country = countries.find(c => c.slug === slug)
  if (!country) return <NotFound />
  
  return <CountryPageContent country={country} />
}
